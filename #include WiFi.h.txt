#include <WiFi.h>
#include <HTTPClient.h>
#include <Wire.h>
#include <DHT.h>
#include <Adafruit_BMP085.h>

// ===== WiFi =====
const char* ssid = "Shakthi’s iPhone";
const char* password = "12345678";

// ===== Server =====
const char* serverName = "http://172.20.10.4:3000/api/data";

// ===== Sensors =====
#define DHTPIN 4
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);

Adafruit_BMP085 bmp;

#define MQ135_PIN 34
#define RAIN_PIN 35

void setup() {
  Serial.begin(115200);
  delay(1000);

  Serial.println("\n🚀 CLIMACORE SYSTEM STARTED\n");

  WiFi.mode(WIFI_STA);
  WiFi.setSleep(false);

  // ===== WiFi Connect =====
  Serial.print("📡 Connecting to WiFi: ");
  Serial.println(ssid);

  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\n✅ WiFi Connected!");
  Serial.print("📍 ESP32 IP: ");
  Serial.println(WiFi.localIP());

  // ===== Sensors =====
  Wire.begin(21, 22);
  dht.begin();

  if (!bmp.begin()) {
    Serial.println("❌ BMP180 NOT detected!");
  } else {
    Serial.println("✅ BMP180 Ready");
  }

  Serial.println("\n🎯 System Ready!\n");
}

void loop() {

  delay(2000);

  Serial.println("\n📊 SENSOR CHECK");

  float temp = dht.readTemperature();
  float hum = dht.readHumidity();
  float pressure = bmp.readPressure();
  int air = analogRead(MQ135_PIN);
  int rain = analogRead(RAIN_PIN);

  // ===== Print =====
  Serial.printf("🌡️ Temp: %.2f °C\n", temp);
  Serial.printf("💧 Humidity: %.2f %%\n", hum);
  Serial.printf("🌪️ Pressure: %.2f Pa\n", pressure);
  Serial.printf("🌫️ Air: %d\n", air);
  Serial.printf("🌧️ Rain: %d\n", rain);

  // ===== CHECK SERVER REACHABILITY =====
  WiFiClient client;

  Serial.print("🌐 Testing server connection... ");

  if (!client.connect("172.20.10.4", 3000)) {
    Serial.println("❌ FAILED (Server unreachable)");
    delay(5000);
    return;
  }

  Serial.println("✅ SUCCESS");
  client.stop();

  // ===== SEND DATA =====
  HTTPClient http;

  http.begin(serverName);
  http.setTimeout(5000);   // 🔥 important fix
  http.addHeader("Content-Type", "application/json");

  String json = "{";
  json += "\"temperature\":" + String(temp) + ",";
  json += "\"humidity\":" + String(hum) + ",";
  json += "\"pressure\":" + String(pressure) + ",";
  json += "\"air\":" + String(air) + ",";
  json += "\"rain\":" + String(rain);
  json += "}";

  Serial.println("📦 Sending JSON:");
  Serial.println(json);

  int code = http.POST(json);

  Serial.print("📨 Response Code: ");
  Serial.println(code);

  if (code > 0) {
    String response = http.getString();
    Serial.println("📥 Server Response:");
    Serial.println(response);
  } else {
    Serial.print("❌ HTTP Error: ");
    Serial.println(http.errorToString(code));
  }

  http.end();

  Serial.println("\n⏳ Waiting...\n");
  delay(5000);
}
